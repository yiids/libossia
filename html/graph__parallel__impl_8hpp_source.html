147  void run(taskflow& tf)
148  {
149  m_tf = &tf;
150  if(tf.m_tasks.empty())
151  {
152  return;
153  }
154 
155  m_toDoTasks = tf.m_tasks.size();
156  m_doneTasks.store(0, std::memory_order_relaxed);
157 
158  for(auto& task : tf.m_tasks)
159  {
160  task.m_remaining_dependencies.store(
161  task.m_dependencies, std::memory_order_relaxed);
162  task.m_executed.store(false, std::memory_order_relaxed);
163 #if defined(CHECK_EXEC_COUNTS)
164  m_checkVec[task.m_taskId] = 0;
165 #endif
166  }
167 
168  std::atomic_thread_fence(std::memory_order_seq_cst);
169 #if defined(DISABLE_DONE_TASKS)
170  ossia::small_pod_vector<ossia::task*, 8> toCleanup;
171 #endif
172  for(auto& task : tf.m_tasks)
173  {
174  if(task.m_dependencies == 0)
175  {
176 #if defined(DISABLE_DONE_TASKS)
177  if(task.m_node->enabled())
178 #endif
179  {
180 #if defined(CHECK_EXEC_COUNTS)
181  m_checkVec[task.m_taskId]++;
182  if(m_checkVec[task.m_taskId] != 1)
183  {
184  fmt::print(
185  stderr, "!!! task {} enqueued {}\n", task.m_taskId,
186  m_checkVec[task.m_taskId]);
187  std::abort();
188  }
189  assert(task.m_dependencies == 0);
190 #endif
191  std::atomic_thread_fence(std::memory_order_release);
192  m_tasks.enqueue(&task);
193  }
194 #if defined(DISABLE_DONE_TASKS)
195  else
196  {
197  toCleanup.push_back(&task);
198  }
199 #endif
200  }
201  }
202 
203 #if defined(DISABLE_DONE_TASKS)
204  for(auto& task : toCleanup)
205  {
206  process_done(*task);
207  }
208 #endif
209 
210  while(m_doneTasks.load(std::memory_order_relaxed) != m_toDoTasks)
211  {
212  task* t{};
213  if(m_tasks.wait_dequeue_timed(t, 1))
214  {
215  execute(*t);
216  }
217  }
218

223  void process_done(ossia::task& task)
224  {
225  if(task.m_executed.exchange(true))
226  return;
227  assert(this->m_doneTasks != m_tf->m_tasks.size());
228 
229 #if defined(DISABLE_DONE_TASKS)
230  ossia::small_pod_vector<ossia::task*, 8> toCleanup;
231 #endif
232  for(int taskId : task.m_precedes)
233  {
234  auto& nextTask = m_tf->m_tasks[taskId];
235  assert(!nextTask.m_executed);
236 
237  std::atomic_int& remaining = nextTask.m_remaining_dependencies;
238  assert(remaining > 0);
239  const int rem = remaining.fetch_sub(1, std::memory_order_relaxed) - 1;
240  assert(rem >= 0);
241  if(rem == 0)
242  {
243 #if defined(DISABLE_DONE_TASKS)
244  if(nextTask.m_node->enabled())
245 #endif
246  {
247 #if defined(CHECK_EXEC_COUNTS)
248  m_checkVec[nextTask.m_taskId]++;
249  if(m_checkVec[nextTask.m_taskId] != 1)
250  {
251  fmt::print(
252  stderr, "!!! task {} enqueued {}\n", nextTask.m_taskId,
253  m_checkVec[nextTask.m_taskId]);
254  std::abort();
255  }
256 #endif
257  std::atomic_thread_fence(std::memory_order_release);
258  m_tasks.enqueue(&nextTask);
259  }
260 #if defined(DISABLE_DONE_TASKS)
261  else
262  {
263  toCleanup.push_back(&nextTask);
264  }
265 #endif
266  }
267  }
268 
269 #if defined(DISABLE_DONE_TASKS)
270  for(auto& clean : toCleanup)
271  {
272  process_done(*clean);
273  }
274 #endif
275 
276  this->m_doneTasks.fetch_add(1, std::memory_order_relaxed);
277  }

283  {

290  }
291 #endif
292 
293 #if defined(CHECK_EXEC_COUNTS)
294  assert(m_checkVec[task.m_taskId] == 1);
295 #endif
296  m_func(*task.m_node);
297 
298 #if defined(CHECK_EXEC_COUNTS)
299  assert(m_checkVec[task.m_taskId] == 1);
300 #endif
301  }
302  catch(...)
303  {
304  fmt::print(stderr, "error !\n");
305  }
306  std::atomic_thread_fence(std::memory_order_release);
307 
308  process_done(task);

310  assert(m_checkVec[task.m_taskId] == 1);
311 #endif
312 
313  std::atomic_thread_fence(std::memory_order_release);
314  }
